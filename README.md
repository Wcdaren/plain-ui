# Install


```
npm i plain-ui -s
```

---

# Quick Start

```js
import Vue from 'vue'
import PlainUI from 'plain-ui/dist/plain-ui.umd'
import 'plain-ui/dist/plain-ui.css'

Vue.use(PlainUI)
```
